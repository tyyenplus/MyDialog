javascript原生移动端优先对话框插件

示例展示

插件配置

var defaults = {
	// 弹框类型
    type : 'alert',     // alert、confirm、info
	// 弹框的title
	titleText : '信息提示',
	// 是否显示title
    showTitle : true,
	// 弹框的内容文本或html文本
    contentHtml : '',
	// 弹框的类名
    dialogClass : '',
	// 自动关闭，false为不自动关闭，给时间数值为多少秒后自动关闭，如3000代表3秒后自动关闭
    autoClose : false,
	// 点击遮罩关闭弹框
    maskClose : false,
	// 拖动
    drag : false,
	// 按钮的文本内容
    buttonText : {
        ok : '确定',
        cancel : '取消',
        delete : '删除'
    },
	// 按钮的类名
    buttonClass : {
        ok : '',
        cancel : '',
        delete : ''
    },
	// 弹框为info时的文本
    infoText : '',      // working in info type
	// 弹框为info时的图标
    infoIcon : '',      // working in info type
	// 点击确定按钮的事件
    onClickOk : function(){},
	// 点击取消按钮的事件
    onClickCancel : function(){},
	// 点击关闭按钮的事件
    onClickClose : function(){},

	// 显示弹框之前执行的事件
    onBeforeShow : function(){},
	// 显示弹框时的事件
    onShow : function(){},
	// 关闭弹框之前的事件
    onBeforeClosed : function(){},
	// 关闭弹框时的事件
    onClosed : function(){}
};

Comfirm 类型, 状态回调函数

$('#btn-08').click(function(){
    new dialog({
       type : 'confirm',
       onBeforeShow : function(){
           alert('显示前执行~~')
       },
       onShow : function(){
           alert('显示完成后执行~~')
       },
       onBeforeClosed : function(){
           alert('关闭前执行~~')
       },
       onClosed : function(){
           alert('关闭后执行，可以看页面title是否改变~~');
           document.title = '我已经关闭拉！';
       },
       contentHtml : '<p>我是confirm类型的对话框。</p> <p>我只是用来占位的内容展示，仅仅用来占位撑起提示内容的高度。我只是用来占位的内容展示，仅仅用来占位撑起提示内容的高度。</p>'
    });
});

info 类型, 更改状态

$('#btn-12').click(function(){
	var infoDialog = new dialog({
	    type : 'info',
	    infoText : '加载中…',
	    infoIcon : 'images/icon/loading.gif'
	});

	window.setTimeout(function() {
		infoDialog.update({
			type: 'confirm',
			autoClose :15000,
			infoText : '操作成功',
			infoIcon : 'images/icon/success.png'
		});
	}, 2500);
});
